    <div id="app">


        <!-- 对象遍历 -->
        <div v-for="(value, key, index) in object">

        </div>

        <!-- 数组对象遍历 -->
        <div v-for="(value, key, index) in objectArray">
            姓名：{{value.name}}
            年龄:{{value.age}}
        </div>

    </div>

    <script type="text/javascript">
        var app = new Vue({
            el: '#app',
            data: {
                // 对象
                object: {
                    name: '张三',
                    names: '李四'
                },

                // 数组对象
                objectArray: [
                    {
                        name: '张三',
                        age: 14
                    },
                    {
                        name: '李四',
                        age: 18
                    }
                ]
            }
        })
    </script>
